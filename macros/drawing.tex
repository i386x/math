\def\setcolor #1({%
  \strip{#1}\@tempa%
  \applyw{setcolor@\@tempa}%
}%
\def\setcolor@gray #1);{%
  \strip{#1}\@tempa%
  \@setcolor@gray\@tempa%
}%
\def\setcolor@rgb #1,#2,#3);{%
  \strip{#3}\@tempc%
  \strip{#2}\@tempb%
  \strip{#1}\@tempa%
  \@setcolor@rgb\@tempa\@tempb\@tempc%
}%
\def\setcolor@cmyk #1,#2,#3,#4);{%
  \strip{#4}\@tempd%
  \strip{#3}\@tempc%
  \strip{#2}\@tempb%
  \strip{#1}\@tempa%
  \@setcolor@cmyk\@tempa\@tempb\@tempc\@tempd%
}%
\def\resetcolor{\setcolor gray(0);}%
%
\def\@setcolor@gray #1{}%
\def\@setcolor@rgb #1#2#3{}%

\font\coverfont=NotoSans-Bold-tlf-t1 at 62pt\relax%
\font\coversubfont=ntx-Italic-tlf-t1 at 25pt\relax%
\def\frontcover{%
  \begingroup%
    \fullpage%
    \setbox0=\hbox{\coverfont Math Notes}%
    \dimen0=\hsize%
    \advance\dimen0-\wd0 %
    \dimen1=62pt %
    \vbox to\paperheight{%
      \pdfliteral{0.5 0.5 0 0 k}%
      \rlap{\vrule width\paperwidth height\paperheight depth 0pt\relax}%
      \kern -.75\vsize\relax%
      \moveright .5\dimen0\vbox{%
        \pdfliteral{1 g 1 G}%
        \hrule width\wd0 height 1.6pt depth 0pt\relax%
        \kern 5.6pt\relax%
        \hrule width\wd0 height 4.8pt depth 0pt\relax%
        \kern .125\dimen1 %
        \copy0 %
        \kern 2\dp0 %
        \kern .125\dimen1 %
        \hrule width\wd0 height 4.8pt depth 0pt\relax%
        \kern 5.6pt\relax%
        \hrule width\wd0 height 1.6pt depth 0pt\relax%
        \kern .125\dimen1%
        \hbox to\wd0{\coversubfont\hss from Mathematical Logic}%
        \pdfliteral{0 g 0 G}%
      }%
      \vfill%
      \pdfliteral{0 g}%
    }%
    \eject%
  \endgroup%
}%
